    frm.add_custom_button("Print", () => {
        let doc = frm.doc;

        // --- YOUR CONDITIONS ---
        let selected_format = null;

        if (doc.my_condition_field === "A") {
            selected_format = "Print Format A";
        } else if (doc.my_condition_field === "B") {
            selected_format = "Print Format B";
        } else {
            selected_format = "Print Format C";   // fallback
        }

        // --- OPEN PRINT PREVIEW DIRECTLY ---
        frappe.route_options = {
            doctype: frm.doctype,
            name: frm.docname,
            format: selected_format,
            no_letterhead: 0
        };

        frappe.set_route("print", {
            doctype: frm.doctype,
            name: frm.docname
        });
    });
}
