Hi @girumtibebu,

That for, please apply the client script for Quotation.

frappe.ui.form.on('Quotation',  {
    refresh: function(frm) {
        frm.set_query("item_code", "items", function (doc, cdt, cdn) {
          return {
            "filters": {
              "item_group": "Services"
            },
          };
        });
    }
});
